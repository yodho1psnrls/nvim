	template <typename, bool>
	friend class IterableToIterator;

	template <bool>

      return _it.base().base();
  }

    if constexpr (STACKABLE)
      return _it.base();
    else

template <typename BaseIterType, bool STACKABLE>
inline std::ostream& operator<<(std::ostream& stream, const BaseTensorIterable<BaseIterType, STACKABLE>& iterable) {
  stream << "{";
  for(auto iter = iterable.begin(); iter != iterable.end();) {
    stream << *iter;
    if (++iter != iterable.end())
      stream << ", ";
  }
  stream << "}";

  return stream;
}
